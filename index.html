  <script>
    // 1. Total Manufacturing Cost
    function calcTotalManufacturingCost(form) {
      const m = parseFloat(form.materials.value);
      const l = parseFloat(form.labor.value);
      const o = parseFloat(form.overhead.value);
      document.getElementById('tmc_result').textContent = `Total Manufacturing Cost: $${(m + l + o).toFixed(2)}`;
    }
    // 2. Direct Materials Cost
    function calcDirectMaterialsCost(form) {
      const b = parseFloat(form.begin.value);
      const p = parseFloat(form.purchases.value);
      const e = parseFloat(form.end.value);
      document.getElementById('dmc_result').textContent = `Direct Materials Cost: $${(b + p - e).toFixed(2)}`;
    }
    // 3. Direct Labor Cost
    function calcDirectLaborCost(form) {
      const w = parseFloat(form.wage.value);
      const h = parseFloat(form.hours.value);
      const t = parseFloat(form.taxes.value);
      const b = parseFloat(form.benefits.value);
      document.getElementById('dlc_result').textContent = `Direct Labor Cost: $${(w * h + t + b).toFixed(2)}`;
    }
    // 4. Overhead Allocation Rate
    function calcOverheadAllocation(form) {
      const o = parseFloat(form.overhead.value);
      const a = parseFloat(form.activity.value);
      if (a === 0) {
        document.getElementById('oar_result').textContent = "Error: Activity base cannot be zero.";
        return;
      }
      document.getElementById('oar_result').textContent = `Overhead Allocation Rate: $${(o / a).toFixed(2)}`;
    }
    // 5. Gross Profit & Margin
    function calcGrossProfit(form) {
      const rev = parseFloat(form.revenue.value);
      const cogs = parseFloat(form.cogs.value);
      const gp = rev - cogs;
      const margin = rev !== 0 ? (gp / rev * 100) : 0;
      document.getElementById('gp_result').textContent = `Gross Profit: $${gp.toFixed(2)} | Gross Margin: ${margin.toFixed(2)}%`;
    }
    // 6. Contribution Margin
    function calcContributionMargin(form) {
      const price = parseFloat(form.price.value);
      const vcost = parseFloat(form.vcost.value);
      const cm = price - vcost;
      const ratio = price !== 0 ? (cm / price) : 0;
      document.getElementById('cm_result').textContent = `Contribution Margin: $${cm.toFixed(2)} | Ratio: ${(ratio * 100).toFixed(2)}%`;
    }
    // 7. Break-Even Point
    function calcBreakEven(form) {
      const fixed = parseFloat(form.fixed.value);
      const price = parseFloat(form.price.value);
      const vcost = parseFloat(form.vcost.value);
      const cm = price - vcost;
      if (cm === 0) {
        document.getElementById('be_result').textContent = "Error: Sales price per unit must be greater than variable cost per unit.";
        return;
      }
      const units = fixed / cm;
      const ratio = price !== 0 ? (cm / price) : 0;
      const sales = ratio !== 0 ? (fixed / ratio) : 0;
      document.getElementById('be_result').textContent = `Break-Even (Units): ${units.toFixed(2)} | Break-Even (Sales $): $${sales.toFixed(2)}`;
    }
    // 8. ROI
    function calcROI(form) {
      const profit = parseFloat(form.profit.value);
      const cost = parseFloat(form.cost.value);
      if (cost === 0) {
        document.getElementById('roi_result').textContent = "Error: Cost of investment cannot be zero.";
        return;
      }
      const roi = (profit / cost) * 100;
      document.getElementById('roi_result').textContent = `ROI: ${roi.toFixed(2)}%`;
    }
    // 9. Payback Period
    function calcPayback(form) {
      const cost = parseFloat(form.cost.value);
      const inflow = parseFloat(form.inflow.value);
      if (inflow === 0) {
        document.getElementById('payback_result').textContent = "Error: Annual net cash inflow cannot be zero.";
        return;
      }
      const period = cost / inflow;
      document.getElementById('payback_result').textContent = `Payback Period: ${period.toFixed(2)} years`;
    }
    // 10. NPV
    function calcNPV(form) {
      const initial = parseFloat(form.initial.value);
      const rate = parseFloat(form.rate.value) / 100;
      const flows = form.flows.value.split(',').map(x => parseFloat(x.trim()));
      let npv = -initial;
      for (let t = 0; t < flows.length; t++) {
        npv += flows[t] / Math.pow(1 + rate, t + 1);
      }
      document.getElementById('npv_result').textContent = `NPV: $${npv.toFixed(2)}`;
    }
    // 11. IRR (simple iterative approach)
    function calcIRR(form) {
      const initial = -Math.abs(parseFloat(form.initial.value));
      const flows = form.flows.value.split(',').map(x => parseFloat(x.trim()));
      let guess = 0.1, step = 0.0001, maxIter = 10000, tol = 1e-5, irr = guess;
      for (let i = 0; i < maxIter; i++) {
        let npv = initial;
        for (let t = 0; t < flows.length; t++) {
          npv += flows[t] / Math.pow(1 + irr, t + 1);
        }
        if (Math.abs(npv) < tol) break;
        irr += (npv > 0 ? step : -step);
      }
      document.getElementById('irr_result').textContent = `IRR: ${(irr * 100).toFixed(2)}%`;
    }
    // 12. Accounting Equation
    function calcAccountingEq(form) {
      const l = parseFloat(form.liabilities.value);
      const e = parseFloat(form.equity.value);
      document.getElementById('aeq_result').textContent = `Assets: $${(l + e).toFixed(2)}`;
    }
    // 13. Current Ratio
    function calcCurrentRatio(form) {
      const a = parseFloat(form.assets.value);
      const l = parseFloat(form.liabilities.value);
      if (l === 0) {
        document.getElementById('cratio_result').textContent = "Error: Current liabilities cannot be zero.";
        return;
      }
      document.getElementById('cratio_result').textContent = `Current Ratio: ${(a / l).toFixed(2)}`;
    }
    // 14. Quick Ratio
    function calcQuickRatio(form) {
      const a = parseFloat(form.assets.value);
      const i = parseFloat(form.inventory.value);
      const l = parseFloat(form.liabilities.value);
      if (l === 0) {
        document.getElementById('qratio_result').textContent = "Error: Current liabilities cannot be zero.";
        return;
      }
      document.getElementById('qratio_result').textContent = `Quick Ratio: ${((a - i) / l).toFixed(2)}`;
    }
    // 15. Debt-to-Equity Ratio
    function calcDebtEquity(form) {
      const l = parseFloat(form.liabilities.value);
      const e = parseFloat(form.equity.value);
      if (e === 0) {
        document.getElementById('debt_result').textContent = "Error: Owner's equity cannot be zero.";
        return;
      }
      document.getElementById('debt_result').textContent = `Debt-to-Equity Ratio: ${(l / e).toFixed(2)}`;
    }
    // 16. Average Freight Cost per Unit
    function calcFreight(form) {
      const t = parseFloat(form.total.value);
      const u = parseFloat(form.units.value);
      if (u === 0) {
        document.getElementById('freight_result').textContent = "Error: Number of units cannot be zero.";
        return;
      }
      document.getElementById('freight_result').textContent = `Average Freight Cost per Unit: $${(t / u).toFixed(2)}`;
    }
    // 17. Total Cost Including Tax
    function calcTotalTax(form) {
      const p = parseFloat(form.price.value);
      const t = parseFloat(form.tax.value) / 100;
      document.getElementById('tax_result').textContent = `Total Cost with Tax: $${(p * (1 + t)).toFixed(2)}`;
    }
    // 18. Net Income
    function calcNetIncome(form) {
      const ops = parseFloat(form.ops.value);
      const interest = parseFloat(form.interest.value);
      const tax = parseFloat(form.tax.value);
      document.getElementById('ni_result').textContent = `Net Income: $${(ops + interest - tax).toFixed(2)}`;
    }
    // 19. Operating Expenses
    function calcOperatingExpenses(form) {
      const s = parseFloat(form.sales.value);
      const r = parseFloat(form.rd.value);
      const g = parseFloat(form.ga.value);
      document.getElementById('opx_result').textContent = `Operating Expenses: $${(s + r + g).toFixed(2)}`;
    }
    // 20. COGS
    function calcCOGS(form) {
      const b = parseFloat(form.begin.value);
      const tmc = parseFloat(form.tmc.value);
      const e = parseFloat(form.end.value);
      document.getElementById('cogs_result').textContent = `COGS: $${(b + tmc - e).toFixed(2)}`;
    }
    // 21. Activity-Based Costing
    function calcActivityBased(form) {
      const t = parseFloat(form.total.value);
      const d = parseFloat(form.drivers.value);
      const u = parseFloat(form.used.value);
      if (d === 0) {
        document.getElementById('abc_result').textContent = "Error: Number of cost driver units cannot be zero.";
        return;
      }
      const costPer = t / d;
      const prodCost = costPer * u;
      document.getElementById('abc_result').textContent = `Cost per Activity: $${costPer.toFixed(2)} | Product Cost: $${prodCost.toFixed(2)}`;
    }
  </script>
